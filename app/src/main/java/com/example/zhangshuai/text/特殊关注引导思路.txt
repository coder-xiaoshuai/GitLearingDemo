 private void checkShouldShowSpecialFollowGuide(){
        if (CollectionUtils.isEmpty(followFeedAdapter.getData())){
            return;
        }
        int size = followFeedAdapter.getData().size();
        int positon = -1;
        for (int i = 0; i < size; i++) {
            if (followFeedAdapter.getData().get(i) instanceof ItemFeedHeaderModel){
                positon = i;
                break;
            }
        }
        if (positon!=-1){
            View view = recyclerViewFeedFollow.getRecyclerView().getLayoutManager().findViewByPosition(positon);
            View popupView = ViewUtils.newInstance(R.layout.layout_special_follow_guide_in_list);
            PopupWindow popupWindow = new PopupWindow(popupView, ViewGroup.LayoutParams.WRAP_CONTENT, ViewGroup.LayoutParams.WRAP_CONTENT, true);
            popupWindow.showAsDropDown(view.findViewById(R.id.image_feed_more),ViewUtils.dpToPx(16),30);
        }
    }
